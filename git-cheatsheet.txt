## Production:
git submodule add https://github.com/rpeszek/crud-ex-backend-yesod.git
git commit -am 'added crud-ex-backend-yesod submodule'

git submodule add https://github.com/rpeszek/crud-ex-frontend-elm.git
git commit -am 'added crud-ex-frontend-elm submodule'

## if changes are made to submodules this fails
git push --recurse-submodules=check

## Consumption:

## clone
git clone --recursive https://github.com/rpeszek/typesafe-web-polyglot.git

## pull
git submodule update
